Kode program SubtractionQuizLoop.py membangkitkan lima pertanyaan pengurangan acak. Lakukan revisi terhadap program itu untuk membangkitkan 10 pertanyaan penambahan acak terhadap dua bilangan bulat antara 1 dan 15. Tampilkan jumlah yang betul dan waktu testnya.

# nomor 1

import time
import random
correctCount = 0  #Count the number of correct answers
count = 0 #Count the number of questions
startTime = time.time()*1000.0
while (count < 10):
    number1 = random.randint(1,16)
    number2 = random.randint(1,16)
    #Prompt the student to answer "what is number1 – number2?"
    print("Berapakah  {}  -  {} ? ".format(number1,number2))
    answer = int(input())
    #Grade the answer and display the result
    if (number1 - number2 == answer):
        print( "Jawaban mu benar!")
        correctCount=correctCount+1
    else:
        print("jawabanmu salah.\n {} - {}  yang benar adalah : ".format(number1,number2), (number1 - number2))
    #Increase the count by incremeting variable called "count"
    count=count+1
print("jawaban yang benar : ",correctCount)
endTime = time.time()*1000.0
print("Waktumu adalah  {} detik".format((endTime - startTime) / 1000))

while derajat<=360:

    cos=math.cos(math.radians(derajat))

# nomor 6

hasil = ''

    if x % 30 == 0:
        continue
    if x % 5 == 0:
        hasil = hasil + str(x) + ' '
        bilangan = bilangan + 1
    elif x % 6 == 0:
        hasil = hasil + str(x) + ' '
        bilangan = bilangan + 1
    
    if bilangan == 10:
        print(hasil)
        bilangan = 0
        hasil = ''
    elif x == 200:
        print(hasil)
