namaFile = input("masukkan nama file: ")
n = len(namaFile)-1
ext = ""
result = ""

while namaFile[n] != ".":
	ext += namaFile[n]
	n -= 1

for i in range(len(ext)-1,-1,-1):
	result += ext[i]

print("nama file", namaFile)	
print("extensinya adalah", result)
